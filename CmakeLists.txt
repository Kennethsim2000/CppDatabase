cmake_minimum_required(VERSION 3.10)
project(cpp_database LANGUAGES CXX)

option(ENABLE_TSAN  "Enable ThreadSanitizer" OFF)
option(ENABLE_ASAN  "Enable AddressSanitizer" OFF)
option(ENABLE_UBSAN "Enable UndefinedBehaviorSanitizer" OFF)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_library(db_core
    src/buffer_pool.cpp
    src/disk_manager.cpp
    src/page.cpp
    src/lru_replacer.cpp
)

add_executable(databasecpp
    src/main.cpp
)

set(SANITIZER_FLAGS)

if (ENABLE_TSAN)
    message(STATUS "ThreadSanitizer enabled")
    list(APPEND SANITIZER_FLAGS
        -fsanitize=thread
        -g
        -O1
    )
endif()

if (ENABLE_ASAN)
    message(STATUS "AddressSanitizer enabled")
    list(APPEND SANITIZER_FLAGS
        -fsanitize=address
        -g
        -O1
    )
endif()

if (ENABLE_UBSAN)
    message(STATUS "UndefinedBehaviorSanitizer enabled")
    list(APPEND SANITIZER_FLAGS
        -fsanitize=undefined
    )
endif()

target_link_libraries(databasecpp PRIVATE db_core)

if (SANITIZER_FLAGS)
    target_compile_options(databasecpp PUBLIC ${SANITIZER_FLAGS})
    target_link_options(databasecpp PUBLIC ${SANITIZER_FLAGS})
endif()
